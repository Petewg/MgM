DECLARE (DLL): Declares a DLL Function.

                    DECLARE <Type> [ STATIC ] <FunctionName> ( [ <type1> <uParam1> ] [, <typeN> <uParamN> ] ) IN <DllName> ALIAS <alias> [FLAGS <flags>]

DLL_TYPE_UINT

DLL_TYPE_DWORD

DLL_TYPE_COLORREF
DLL_TYPE_BYTE

DLL_TYPE_HWND

DLL_TYPE_HGLOBAL
DLL_TYPE_WPARAM
DLL_TYPE_HKEY
DLL_TYPE_CHAR

DLL_TYPE_WCHAR

DLL_TYPE_LPSTR
DLL_TYPE_ULONG
